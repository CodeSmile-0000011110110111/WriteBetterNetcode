99 private static Dictionary<String, String> ParseCmdLineArgs()
100 {
101 var parsedArgs = new Dictionary<String, String>();
102 var cmdArgs = Environment.GetCommandLineArgs();
103 var argCount = cmdArgs.Length;
104
105 // make all lowercase
106 for (var i = 0; i < argCount; ++i)
107 cmdArgs[i] = cmdArgs[i].ToLower();
108
109 // parse
110 for (var i = 0; i < argCount; ++i)
111 {
112 var argKey = cmdArgs[i];
113 if (argKey.StartsWith("-"))
114 {
115 // last argKey may not have a value
116 var argValue = i < argCount - 1 ? cmdArgs[i + 1] : null;
117
118 // don't use the next argKey as the value
119 argValue = argValue?.StartsWith("-") ?? false ? null : argValue;
120
121 // remove the leading argKey dash
122 argKey = argKey.Substring(1, argKey.Length - 1);
123
124 if (parsedArgs.ContainsKey(argKey) == false)
125 parsedArgs.Add(argKey, argValue);
126 else if (Application.isEditor == false)
127 Debug.LogWarning($"Duplicate argument: -{argKey}");
128 }
129 }
130
131 return parsedArgs;
132 }
133 }
