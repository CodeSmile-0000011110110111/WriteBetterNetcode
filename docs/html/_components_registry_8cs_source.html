18 [DisallowMultipleComponent]
19 public sealed class ComponentsRegistry : MonoBehaviour
20 {
21 public static event Action<Type, Component> OnComponentAssigned;
22
23 private static Dictionary<Type, Component> s_Components;
24 private static Dictionary<Type, List<TaskCompletionSource<Object>>> s_AwaitingAssignmentSources;
25
34 public static T Get<T>() where T : Component =>
35 s_Components.ContainsKey(typeof(T)) ? s_Components[typeof(T)] as T : null;
36
45 public static async Task<T> GetAsync<T>() where T : Component
46 {
47 // if already assigned, return instantly
48 var component = Get<T>();
49 if (component != null)
50 return component;
51
52 // create entry for type in awaitables
53 if (s_AwaitingAssignmentSources.ContainsKey(typeof(T)) == false)
54 s_AwaitingAssignmentSources[typeof(T)] = new List<TaskCompletionSource<Object>>();
55
56 // add a new completion source
57 var tcs = new TaskCompletionSource<Object>();
58 s_AwaitingAssignmentSources[typeof(T)].Add(tcs);
59
60 // await here so the result can be cast to T
61 return await tcs.Task.ConfigureAwait(false) as T;
62 }
63
72 public static void Set<T>(T component) where T : Component
73 {
74 s_Components[typeof(T)] = component;
75 OnComponentAssigned?.Invoke(typeof(T), component);
76 ProcessAssignmentAwaitables(component);
77 }
78
79 private static void ProcessAssignmentAwaitables<T>(T component) where T : Component
80 {
81 if (s_AwaitingAssignmentSources.TryGetValue(typeof(T), out var awaitables))
82 {
83 foreach (var completionSource in awaitables)
84 completionSource.SetResult(component);
85
86 s_AwaitingAssignmentSources.Remove(typeof(T));
87 }
88 }

110 private void LogComponents()
111 {
112 var sb = new StringBuilder("ComponentsRegistry contains:\n");
113 if (s_Components != null)
114 {
115 sb.AppendLine($"{s_Components.Count} components:");
116 foreach (var kvp in s_Components)
117 {
118 var type = kvp.Key;
119 var component = kvp.Value;
120 sb.AppendLine($"<b>{type.Name}</b> (GO: '{component?.name}', ID: {component?.GetInstanceID()})");
121 }
122 }
123 else
124 sb.AppendLine("<no components>");
125
126 Debug.Log(sb.ToString());
127 }
128#endif
