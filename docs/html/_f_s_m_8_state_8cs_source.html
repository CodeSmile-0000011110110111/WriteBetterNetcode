96 public State AddTransitions(params Transition[] transitions)
97 {
98 if (transitions == null || transitions.Length == 0)
99 throw new ArgumentException("transitions null or empty");
100
101 if (m_Transitions == null)
102 m_Transitions = transitions;
103 else
104 {
105 var startIndex = m_Transitions.Length;
106 var addCount = transitions.Length;
107 Array.Resize(ref m_Transitions, startIndex + addCount);
108
109 for (var i = 0; i < addCount; i++)
110 m_Transitions[startIndex + i] = transitions[i];
111 }
112
113 return this;
114 }
