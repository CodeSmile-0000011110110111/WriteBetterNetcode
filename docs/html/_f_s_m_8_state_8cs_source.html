53 public State AddTransitions(params Transition[] transitions)
54 {
55 if (transitions == null || transitions.Length == 0)
56 throw new ArgumentException("transitions null or empty");
57
58 if (m_Transitions == null)
59 m_Transitions = transitions;
60 else
61 {
62 var startIndex = m_Transitions.Length;
63 var addCount = transitions.Length;
64 Array.Resize(ref m_Transitions, startIndex + addCount);
65
66 for (var i = 0; i < addCount; i++)
67 m_Transitions[startIndex + i] = transitions[i];
68 }
69
70 return this;
71 }
