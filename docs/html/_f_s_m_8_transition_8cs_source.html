28 internal static Boolean ConditionsSatisfied(FSM sm, String transitionName, ICondition[] conditions, Boolean logging,
29 Boolean useLogicalOr = false)
30 {
31 foreach (var condition in conditions)
32 {
33 var satisfied = condition.IsSatisfied(sm);
34
35 if (logging)
36 LogCondition(sm, transitionName, condition, satisfied);
37
38 // early out checks
39 if (useLogicalOr)
40 {
41 if (satisfied) // OR: if one is true
42 return true;
43 }
44 else if (!satisfied) // AND: if none are false
45 return false;
46 }
47
48 // if we get here: OR = all condition were false / AND = all conditions were true
49 return useLogicalOr ? false : true;
50 }

52 internal static async void ExecuteActions(FSM sm, String transitionName, IAction[] actions, Boolean logging = false)
53 {
54 if (actions == null)
55 return;
56
57 var actionCount = actions.Length;
58 if (actionCount == 0)
59 return;
60
61 for (var i = 0; i < actionCount; i++)
62 {
63 var action = actions[i];
64
65 if (logging)
66 LogExecuteAction(sm, transitionName, action);
67
68 if (action is IAsyncAction asyncAction)
69 await asyncAction.ExecuteAsync(sm);
70 else
71 action.Execute(sm);
72 }
73 }

190 internal void Update(FSM sm)
191 {

207 private void ExecuteActionsWithErrorHandling(FSM sm, IAction[] actions, Boolean logging)
208 {
209 try
210 {
211 ExecuteActions(sm, Name, actions, logging);
212 }
213 catch (Exception e)
214 {
215 if (ErrorActions != null || ErrorGotoState != null)
216 {
217 Debug.LogWarning($"{sm.ActiveState.Name} [{Name}]: ERROR => {e}");
218
219 ExecuteActions(sm, Name, ErrorActions, logging);
220
221 if (ErrorGotoState != null)
222 {
223 if (logging)
224 LogStateChange(sm, ErrorGotoState, Name);
225
226 sm.SetActiveState(ErrorGotoState);
227 }
228 }
229 else
230 throw;
231 }
