25 internal static Boolean ConditionsSatisfied(FSM sm, String transitionName, ICondition[] conditions, Boolean logging,
26 Boolean useLogicalOr = false)
27 {
28 foreach (var condition in conditions)
29 {
30 var satisfied = condition.IsSatisfied(sm);
31
32 if (logging)
33 LogCondition(sm, transitionName, condition, satisfied);
34
35 // early out checks
36 if (useLogicalOr)
37 {
38 if (satisfied) // OR: if one is true
39 return true;
40 }
41 else if (!satisfied) // AND: if none are false
42 return false;
43 }
44
45 // if we get here: OR = all condition were false / AND = all conditions were true
46 return useLogicalOr ? false : true;
47 }

49 internal static async void ExecuteActions(FSM sm, String transitionName, IAction[] actions, Boolean logging = false)
50 {
51 if (actions == null)
52 return;
53
54 var actionCount = actions.Length;
55 if (actionCount == 0)
56 return;
57
58 for (var i = 0; i < actionCount; i++)
59 {
60 var action = actions[i];
61
62 if (logging)
63 LogExecuteAction(sm, transitionName, action);
64
65 if (action is IAsyncAction asyncAction)
66 await asyncAction.ExecuteAsync(sm);
67 else
68 action.Execute(sm);
69 }
70 }

148 internal void Update(FSM sm)
149 {
150 var logging = sm.ActiveState.Logging;
151 if (ConditionsSatisfied(sm, Name, Conditions, logging))
152 {

154

165 private void ExecuteActionsWithErrorHandling(FSM sm, IAction[] actions, Boolean logging)
166 {
167 try
168 {
169 ExecuteActions(sm, Name, actions, logging);
170 }
171 catch (Exception e)
172 {
173 if (ErrorActions != null || ErrorGotoState != null)
174 {
175 Debug.LogWarning($"{sm.ActiveState.Name} [{Name}]: ERROR => {e}");
176
177 ExecuteActions(sm, Name, ErrorActions, logging);
178
179 if (ErrorGotoState != null)
180 {
181 if (logging)
182 LogStateChange(sm, ErrorGotoState, Name);
183
184 sm.SetActiveState(ErrorGotoState);
185 }
186 }
187 else
188 throw;
189 }
