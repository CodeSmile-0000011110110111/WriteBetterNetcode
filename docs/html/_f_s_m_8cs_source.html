178 public void Update()
179 {
180 ThrowIfStatemachineNotStarted();
181
182 if (IsStopped)
183 return;
184
185 var iterations = AllowMultipleStateChanges ? MaxStateChangesPerEvaluate : 0;
186 do
187 {
188 //Debug.Log($"[{Time.frameCount}] FSM Update: iteration #{MaxStateChangesPerEvaluate-iterations}");
189
190 DidChangeState = false;
191
192 var updatingState = ActiveState;
193 updatingState.Update(this);
194
195 if (DidChangeState)
196 {
197 updatingState.OnExitState(this);
198 ActiveState.OnEnterState(this);
199 InvokeOnStateChangeEvent(updatingState);
200
201 if (ActiveState.IsFinalState())
202 {
203 Stop();
204 InvokeOnStoppedEvent();
205 }
206
207 iterations--;
208 }
209
210 if (IsStopped || DidChangeState == false)
211 break;
212
213 if (AllowMultipleStateChanges)
214 WarnIfIterationsExceeded(iterations);
215 } while (iterations > 0);
216 }
