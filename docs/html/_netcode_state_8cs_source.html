88 private void SetupStatemachine()
89 {
90 m_Statemachine = new FSM($"{nameof(NetcodeState)}Machine")
91 .WithStates(Enum.GetNames(typeof(State)));
92
93 m_Statemachine.AllowMultipleStateChanges = true;
94
95 var states = m_Statemachine.States;
96 var initState = states[(Int32)State.Initializing];
97 var offlineState = states[(Int32)State.Offline];
98 var relayStartState = states[(Int32)State.RelayStarting];
99 var networkStartState = states[(Int32)State.NetworkStarting];
100 var serverOnlineState = states[(Int32)State.ServerOnline];
101 var clientOnlineState = states[(Int32)State.ClientOnline];
102 var clientPlayingState = states[(Int32)State.ClientPlaying];
103 var networkStopState = states[(Int32)State.NetworkStopping];
104
105 m_NetcodeConfigVar = m_Statemachine.Vars.DefineVar<NetcodeConfig>();
106 m_RelayConfigVar = m_Statemachine.Vars.DefineVar<RelayConfig>();
107 m_TransportConfigVar = m_Statemachine.Vars.DefineVar<TransportConfig>();
108 var relayInitOnceVar = m_Statemachine.Vars.DefineBool("RelayInitOnce");
109
110 // for testing
111 //m_Statemachine.Logging = true;
112 m_Statemachine.OnStateChange += args => Debug.Log($"<color=olive>[{Time.frameCount}] {m_Statemachine} changed " +
113 $"from {args.PreviousState} to {args.ActiveState}</color>");
114
115 var invokeWentOnline = new LambdaAction($"{nameof(WentOnline)}.Invoke",
116 () => WentOnline?.Invoke(m_NetcodeConfigVar.Value.Role));
117 var invokeWentOffline = new LambdaAction($"{nameof(WentOffline)}.Invoke",
118 () => WentOffline?.Invoke(m_NetcodeConfigVar.Value.Role));
119 var tryInvokeRelayJoinCodeAvailable = new LambdaAction(nameof(TryInvokeRelayJoinCodeAvailable),
120 () => TryInvokeRelayJoinCodeAvailable());
121
122 var resetNetcodeState = new CompoundAction("ResetNetcodeState",
123 new SetFalse(relayInitOnceVar),
124 new SetNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None),
125 new RelayClearAllocationData(m_RelayConfigVar),
126 invokeWentOffline);
127
128 /* TODO (MAJOR ONES)
129 * - forward events (eg join code available)
130 * route through config object ... BUT: config is a struct
131 * - Var<T> allow classes?
132 * - correct error handling for compound actions etc because they call the non-catching T methods!
133 * - support client connection dis-approval (received client disconnect event with reason)
134 */
135
136 // Init state
137 initState.AddTransition("Init Completed")
138 .ToState(offlineState)
139 .WithConditions(new IsNetworkManagerSingletonAssigned());
140
141 // Offline state
142 offlineState.AddTransition("Start with Relay")
143 .ToState(relayStartState)
144 .WithConditions(
145 new IsNotNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None),
146 new IsRelayEnabled(m_RelayConfigVar))
147 .WithActions(invokeWentOnline);
148
149 offlineState.AddTransition("Start w/o Relay")
150 .ToState(networkStartState)
151 .WithConditions(
152 new IsNotNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None),
153 FSM.NOT(new IsRelayEnabled(m_RelayConfigVar)))
154 .WithActions(invokeWentOnline);
155
156 // Relay state
157 relayStartState.AddTransition("Relay Alloc/Join")
158 .WithConditions(new IsFalse(relayInitOnceVar))
159 .WithActions(
160 new SetTrue(relayInitOnceVar),
161 new UnityServicesInit(),
162 new SignInAnonymously(),
163 new RelayCreateOrJoinAllocation(m_NetcodeConfigVar, m_RelayConfigVar))
164 .ToErrorState(offlineState)
165 .WithErrorActions(resetNetcodeState);
166
167 relayStartState.AddTransition("Relay Started")
168 .ToState(networkStartState)
169 .WithConditions(
170 new IsSignedIn(),
171 new IsRelayReady(m_RelayConfigVar))
172 .WithActions(tryInvokeRelayJoinCodeAvailable);
173
174 // NetworkStart state
175 networkStartState.AddTransition("Cancel Net Start")
176 .ToState(offlineState)
177 .WithConditions(new IsNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None))
178 .WithActions(resetNetcodeState);
179
180 networkStartState.AddTransition("Network Starting")
181 .WithConditions(new IsNotListening())
182 .WithActions(
183 new TransportSetup(m_NetcodeConfigVar, m_TransportConfigVar, m_RelayConfigVar),
184 new NetworkStart(m_NetcodeConfigVar))
185 .ToErrorState(offlineState)
186 .WithErrorActions(resetNetcodeState);
187
188 networkStartState.AddTransition("Server started")
189 .ToState(serverOnlineState)
190 .WithConditions(new IsLocalServerStarted());
191 networkStartState.AddTransition("Client started")
192 .ToState(clientOnlineState)
193 .WithConditions(new IsLocalClientStarted());
194
195 // Server States
196 serverOnlineState.AddTransition("Server stopped")
197 .ToState(networkStopState)
198 .WithConditions(
199 FSM.OR(new IsLocalServerStopped(),
200 new IsNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None)))
201 .WithActions(new NetworkStop());
202
203 // Client States
204 clientOnlineState.AddTransition("Client connected")
205 .ToState(clientPlayingState)
206 .WithConditions(new IsLocalClientConnected());
207 clientOnlineState.AddTransition("Client stopped")
208 .ToState(networkStopState)
209 .WithConditions(
210 FSM.OR(new IsLocalClientStopped(),
211 new IsNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None)))
212 .WithActions(new NetworkStop());
213
214 clientPlayingState.AddTransition("Client disconnected or stopped")
215 .ToState(networkStopState)
216 .WithConditions(FSM.OR(
217 new IsLocalClientDisconnected(),
218 new IsLocalClientStopped(),
219 new IsNetcodeRole(m_NetcodeConfigVar, NetcodeRole.None)))
220 .WithActions(new NetworkStop());
221
222 // Stopping state
223 networkStopState.AddTransition("Network stopped")
224 .ToState(offlineState)
225 .WithConditions(new IsNetworkOffline())
226 .WithActions(resetNetcodeState);
227 }

229 private void TryInvokeRelayJoinCodeAvailable()
230 {
231 var role = m_NetcodeConfigVar.Value.Role;
232 if (role == NetcodeRole.Server || role == NetcodeRole.Host)
233 {
234 var joinCode = m_RelayConfigVar.Value.JoinCode;
235 if (String.IsNullOrEmpty(joinCode) == false)
236 RelayJoinCodeAvailable?.Invoke(joinCode);
237 }
238 }

240 public void RequestStart(NetcodeConfig netcodeConfig,
241 TransportConfig transportConfig, RelayConfig relayConfig = default)
242 {
243 if (netcodeConfig.Role == NetcodeRole.None)
244 throw new ArgumentException("cannot start without role");
245
246 Debug.Log(netcodeConfig);
247 Debug.Log(transportConfig);
248 if (relayConfig.UseRelay)
249 Debug.Log(relayConfig);
250
251 m_NetcodeConfigVar.Value = netcodeConfig;
252 m_TransportConfigVar.Value = transportConfig;
253 m_RelayConfigVar.Value = relayConfig;
254 }
255
256 public void RequestStopNetwork()
257 {
258 var netcodeConfig = m_NetcodeConfigVar.Value;
259 netcodeConfig.Role = NetcodeRole.None;
260 m_NetcodeConfigVar.Value = netcodeConfig;
261 }
