175 private void OnSceneEvent(SceneEvent sceneEvent)
176 {
177 var completed = String.Empty;
178 var timedOut = String.Empty;
179
180 var completedCount = sceneEvent.ClientsThatCompleted?.Count;
181 var timedOutCount = sceneEvent.ClientsThatTimedOut?.Count;
182
183 if (completedCount > 0)
184 {
185 var first = true;
186 var sb = new StringBuilder();
187 foreach (var clientId in sceneEvent.ClientsThatCompleted)
188 sb.Append($"{(first ? "" : ", ")}{clientId}");
189
190 completed = $"(completed clientIDs: {sb})";
191 }
192
193 if (timedOutCount > 0)
194 {
195 var first = true;
196 var sb = new StringBuilder();
197 foreach (var clientId in sceneEvent.ClientsThatTimedOut)
198 sb.Append($"{(first ? "" : ", ")}{clientId}");
199
200 timedOut = $"<color=red>(TIMEOUT clientIDs: {sb})<color>";
201 }
202
203 var role = sceneEvent.ClientId == 0 ? "Server" : $"Client {sceneEvent.ClientId}";
204 Log($"{role} Scene: {sceneEvent.SceneEventType} {sceneEvent.SceneName} ({sceneEvent.LoadSceneMode}) " +
205 $"{completed} {timedOut}");
206 }

208 private void Log(String message)
209 {
210 var localTick = -1;
211 var serverTick = -1;
212 var frameCount = Time.frameCount;
213
214 var net = NetworkManager.Singleton;
215 if (net)
216 {
217 localTick = net.LocalTime.Tick;
218 serverTick = net.ServerTime.Tick;
219 }
220
221 var color = $"<color=#{m_LogTextColor.r:x02}{m_LogTextColor.g:x02}{m_LogTextColor.b:x02}>";
222 Debug.Log($"{color}[L:{localTick}|S:{serverTick}|F:{frameCount}] {message}</color>");
223 }
