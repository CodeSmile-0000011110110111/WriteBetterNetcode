93 public void OnValidate()
94 {
95#if UNITY_EDITOR
96 SceneAsset = m_SceneAsset;
97
98 if (m_ScenePath != null)
99 {
100 var existsInBuildScenes = false;
101 var buildScenes = EditorBuildSettings.scenes;
102 foreach (var buildScene in buildScenes)
103 {
104 if (buildScene.enabled && buildScene.path.ToLower() == m_ScenePath.ToLower())
105 {
106 existsInBuildScenes = true;
107 break;
108 }
109 }
110
111 // TODO: auto-add to build index? should have an Inspector button for that
112 if (existsInBuildScenes == false)
113 Debug.LogWarning($"FYI: Scene '{m_SceneName}' is not in the build index yet or inactive.");
114 }
115#endif
