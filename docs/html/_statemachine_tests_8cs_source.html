51 public void FSM_StartEndStates_RunsToEndAndRaisesStateChangeEvent()
52 {
53 var didExecuteActions = false;
54
55 var sm = new FSM("FSM").WithStates("START", "END");
56 sm.States[0].AddTransition().ToState(sm.States[1]).WithActions(FSM.Action(() => didExecuteActions = true));
57
58 sm.Start().Update();
59
60 Assert.AreEqual(sm.States[1], sm.ActiveState);
61 Assert.IsTrue(sm.IsStopped);
62 Assert.IsTrue(didExecuteActions);
63 }

65 [Test]
66 public void FSM_StateChange_InvokesStateChangeEvent()
67 {
68 var sm = new FSM("FSM").WithStates("START", "END");
69 sm.States[0].AddTransition().ToState(sm.States[1]);
70
71 var didInvokeStateChangedEvent = false;
72 sm.OnStateChange += args =>
73 {
74 didInvokeStateChangedEvent = true;
75 Assert.AreEqual(args.PreviousState, sm.States[0]);
76 Assert.AreEqual(args.ActiveState, sm.States[1]);
77 };
78
79 sm.Start().Update();
80
81 Assert.IsTrue(didInvokeStateChangedEvent);
82 }
