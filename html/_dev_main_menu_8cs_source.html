30 private static void RequestStart(NetcodeConfig netcodeConfig,
31 TransportConfig transportConfig, RelayConfig relayConfig)
32 {
33 var netcodeState = ComponentsRegistry.Get<NetcodeState>();
34 Debug.Log($"netcodeState: {netcodeState}");
35
36 netcodeState.RequestStart(netcodeConfig, transportConfig, relayConfig);

51 private void Start() => RegisterNetcodeStateEvents();

57 private void RegisterNetcodeStateEvents()
58 {
59 var netcodeState = ComponentsRegistry.Get<NetcodeState>();
60 netcodeState.WentOnline += WentOnline;
61 netcodeState.WentOffline += WentOffline;
62
63 // we may already be online eg via MPPM
64 if (netcodeState.IsOnline)
65 Hide();
66 }
67
68 private void UnregisterNetcodeStateEvents()
69 {
70 var netcodeState = ComponentsRegistry.Get<NetcodeState>();
71 if (netcodeState != null)
72 {
73 netcodeState.WentOnline -= WentOnline;
74 netcodeState.WentOffline -= WentOffline;
75 }
76 }

78 private void WentOnline(NetcodeRole role) => Hide();
79 private void WentOffline(NetcodeRole role) => Show();

103 private void OnJoinCodeChanged(ChangeEvent<String> evt)
104 {
105 JoinCodeField.SetValueWithoutNotify(evt.newValue.ToUpper());
106
107 var textColor = Color.black;
108 if (String.IsNullOrWhiteSpace(evt.newValue) ||
109 evt.newValue.Length != 6)
110 textColor = Color.red;
111
112 JoinCodeField.style.color = textColor;
113 }
114
115 private void OnAddressFieldChanged(ChangeEvent<String> evt)
116 {
117 var textColor = Color.black;
118 if (!IPAddress.TryParse(evt.newValue, out var _))
119 textColor = Color.red;
120
121 AddressField.style.color = textColor;
122 }
123
124 private void OnPortFieldChanged(ChangeEvent<String> evt)
125 {
126 var textColor = Color.black;
127 if (!UInt16.TryParse(evt.newValue, out var port) ||
128 port < 1024)
129 textColor = Color.red;
130
131 PortField.style.color = textColor;
132 }

134 private void OnHostRelayButtonClicked() => StartHost(true);
135 private void OnHostDirectButtonClicked() => StartHost(false);
136 private void OnJoinRelayButtonClicked() => JoinWithRelay(JoinCodeField.text);
137 private void OnJoinDirectButtonClicked() => JoinWithAddress(AddressField.text, PortField.text);

139 private void StartHost(Boolean withRelay)
140 {
141 var netcodeConfig = NetcodeConfig.FromCmdArgs();
142 netcodeConfig.Role = NetcodeRole.Host;
143
144 var transportConfig = TransportConfig.FromNetworkManagerWithCmdArgOverrides();
145 transportConfig.ServerListenAddress = "0.0.0.0";
146 transportConfig.UseWebSockets = AllowWebClientsToggle.value;
147
148 var relayConfig = RelayConfig.FromCmdArgs();
149 relayConfig.UseRelay = withRelay;
150
151 RequestStart(netcodeConfig, transportConfig, relayConfig);
152 }
153
154 private void JoinWithRelay(String joinCode)
155 {
156 var netcodeConfig = NetcodeConfig.FromCmdArgs();
157 netcodeConfig.Role = NetcodeRole.Client;
158
159 var transportConfig = TransportConfig.FromNetworkManagerWithCmdArgOverrides();
160
161 var relayConfig = RelayConfig.FromCmdArgs();
162 relayConfig.UseRelay = true;
163 relayConfig.JoinCode = joinCode;
164
165 RequestStart(netcodeConfig, transportConfig, relayConfig);
166 }
167
168 private void JoinWithAddress(String address, String portStr)
169 {
170 var netcodeConfig = NetcodeConfig.FromCmdArgs();
171 netcodeConfig.Role = NetcodeRole.Client;
172
173 var transportConfig = TransportConfig.FromNetworkManagerWithCmdArgOverrides();
174 transportConfig.Address = address;
175 if (UInt16.TryParse(portStr, out var port))
176 transportConfig.Port = port;
177
178 var relayConfig = RelayConfig.FromCmdArgs();
179
180 RequestStart(netcodeConfig, transportConfig, relayConfig);
181 }
182 }
