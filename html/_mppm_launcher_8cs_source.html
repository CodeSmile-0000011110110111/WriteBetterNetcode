4using CodeSmile.Components.Registry;
5using CodeSmile.Statemachine.Netcode;
6using System;
7using System.Linq;
8using Unity.Multiplayer.Playmode;
9using UnityEditor;
10using UnityEngine;
11
12namespace CodeSmile.MultiPal.Netcode
13{
14 [DisallowMultipleComponent]
15 internal sealed class MppmLauncher : MonoBehaviour
16 {
17#if UNITY_EDITOR
18 private static Boolean m_DidLaunchOnce;
19
20 private void Start()
21 {
22 if (m_DidLaunchOnce)
23 return;
24
25 m_DidLaunchOnce = true;
26
27 var role = GetNetworkRoleFromMppmTags();
28 if (role != NetcodeRole.None)
29 StartNetworkWithRole(role);
30 }
31
32 private static NetcodeRole GetNetworkRoleFromMppmTags()
33 {
34 var playerTags = CurrentPlayer.ReadOnlyTags();
35 var roleCount = Enum.GetValues(typeof(NetcodeRole)).Length;
36
37 for (var roleIndex = 0; roleIndex < roleCount; roleIndex++)
38 {
39 if (playerTags.Contains(((NetcodeRole)roleIndex).ToString()))
40 return (NetcodeRole)roleIndex;
41 }
42
43 return NetcodeRole.None;
44 }
45
46 private static void StartNetworkWithRole(NetcodeRole role)
47 {
48 var netcodeConfig = new NetcodeConfig { Role = role, MaxConnections = 4 };
49 var transportConfig = TransportConfig.FromNetworkManager();
50
51 var netcodeState = ComponentsRegistry.Get<NetcodeState>();
52 netcodeState.RequestStart(netcodeConfig, transportConfig);
53 }
54
55 // ensure unsaved Material, ScriptableObject, etc changes are applied to virtual players
56 [InitializeOnLoadMethod]
57 private static void InitOnLoad() => EditorApplication.playModeStateChanged += state =>
58 {
59 if (state == PlayModeStateChange.ExitingEditMode)
60 SaveProject();
61 };
62
63 [RuntimeInitializeOnLoadMethod(RuntimeInitializeLoadType.SubsystemRegistration)]
64 private static void ResetStaticFields() => m_DidLaunchOnce = false;
65
66 private static void SaveProject() => AssetDatabase.SaveAssets();
67#endif
68 }
69}
