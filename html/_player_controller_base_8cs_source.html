146 private void TryMoveCharacterControllerToTarget(Transform target)
147 {
148 CharController = null;
149
150 // check if we are using the CharacterController component
151 var ourCharCtrl = GetComponent<CharacterController>();
152 if (ourCharCtrl == null)
153 return;
154
155 // if CharCtrl present on target, use that, otherwise add one
156 if (!target.TryGetComponent<CharacterController>(out var targetCharCtrl))
157 targetCharCtrl = target.gameObject.AddComponent<CharacterController>();
158
159 CopyInspectorProperties(ourCharCtrl, targetCharCtrl);
160 ourCharCtrl.enabled = false; // make sure the source ctrl doesn't get in the way
161
162 CharController = targetCharCtrl;
163 }
