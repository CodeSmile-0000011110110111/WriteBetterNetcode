62 public void OnValidate()
63 {
64#if UNITY_EDITOR
65 SceneAsset = m_SceneAsset;
66
67 if (m_ScenePath != null)
68 {
69 var existsInBuildScenes = false;
70 var buildScenes = EditorBuildSettings.scenes;
71 foreach (var buildScene in buildScenes)
72 {
73 if (buildScene.enabled && buildScene.path.ToLower() == m_ScenePath.ToLower())
74 {
75 existsInBuildScenes = true;
76 break;
77 }
78 }
79
80 // TODO: auto-add to build index? should have an Inspector button for that
81 if (existsInBuildScenes == false)
82 Debug.LogWarning($"FYI: Scene '{m_SceneName}' is not in the build index yet or inactive.");
83 }
84#endif
