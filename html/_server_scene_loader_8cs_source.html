143 var scene = EngineSceneManager.GetSceneByPath(sceneReference.ScenePath);
144 if (scene.IsValid() && scene.isLoaded)

161 private async Task InternalLoadUnloadScenesAsync(SceneReference[] scenes, SceneOperation operation)
162 {
163 if (m_IsServerOnline == false) // also guards against clients calling into here
164 return;
165
166 if (scenes == null || scenes.Length == 0)
167 return;
168
169 if (IsProcessingEvents)
170 {
171 // can't load again while still loading/unloading!
172 // Note: while possible to add scenes to ongoing queue, there may be different callers awaiting so we'd force
173 // one caller to also wait for the other's scenes! This is unexpected with possible severe side-effects!
174 Debug.LogError("Scene load/unload event in progress! Forgot to await??");
175 return;
176 }
177
178 // task is for awaiting, will get its result when entire queue has been processed
179 m_TaskInProgress = new TaskCompletionSource<Boolean>();
180 m_ScenesProcessing = new Queue<ProcessingScene>();
181 EnqueueScenes(scenes, operation);
182
183 ProcessNextSceneInQueue();
184
185 // await our dummy result until queue is complete
186 await m_TaskInProgress.Task;
187 m_TaskInProgress = null; // we no longer need you
188 }
189
190 private void EnqueueScenes(SceneReference[] scenes, SceneOperation operation)
191 {
192 foreach (var sceneRef in scenes)
193 m_ScenesProcessing.Enqueue(new ProcessingScene { SceneRef = sceneRef, Operation = operation });
194 }
195
196 private void InternalLoadQueuedSceneAsync(SceneReference sceneRef)
197 {
198 if (m_SynchedScenes.Contains(sceneRef))
199 {
200 //Debug.LogWarning($"skip load, already loaded: {sceneRef.SceneName}");
201 ProcessNextSceneInQueue();
202 return;
203 }
204
205 var status = SceneManager.LoadScene(sceneRef.SceneName, LoadSceneMode.Additive);
206
207 if (status == SceneEventProgressStatus.Started)
208 m_SynchedScenes.Add(sceneRef);
209 else
210 {
211 Debug.LogWarning($"Scene load failed: {status} for {sceneRef.ScenePath}");
212 ProcessNextSceneInQueue();
213 }
214 }
215
216 private void InternalUnloadQueuedSceneAsync(SceneReference sceneRef)
217 {
218 if (m_SynchedScenes.Contains(sceneRef) == false)
219 {
220 //Debug.LogWarning($"skip unload, not loaded: {sceneRef.SceneName}");
221 ProcessNextSceneInQueue();
222 return;
223 }
224
225 var status = SceneManager.UnloadScene(sceneRef.RuntimeScene);
226
227 if (status == SceneEventProgressStatus.Started)
228 m_SynchedScenes.Remove(sceneRef);
229 else
230 {
231 Debug.LogWarning($"Scene unload failed: {status} for {sceneRef.ScenePath}");
232 ProcessNextSceneInQueue();
233 }
234 }
235
236 private void ProcessNextSceneInQueue()
237 {
238 if (m_ScenesProcessing.Count > 0)
239 {
240 // process the next scene
241 var processingScene = m_ScenesProcessing.Dequeue();
242 if (processingScene.Operation == SceneOperation.Load)
243 InternalLoadQueuedSceneAsync(processingScene.SceneRef);
244 else
245 InternalUnloadQueuedSceneAsync(processingScene.SceneRef);
246 }
247 else
248 {
249 // end processing, SetResult stops awaiting
250 m_ScenesProcessing = null;
251 m_TaskInProgress.SetResult(true);
252 }
253 }

255 private void OnServerSceneEvents(SceneEvent sceneEvent)
256 {
257 switch (sceneEvent.SceneEventType)
258 {
259 case SceneEventType.Load:
260 case SceneEventType.Unload:
261 sceneEvent.AsyncOperation.completed += OnServerSceneEventComplete;
262 break;
263 }
264 }
265
266 private void OnServerSceneEventComplete(AsyncOperation asyncOp)
267 {
268 asyncOp.completed -= OnServerSceneEventComplete;
269
270 ProcessNextSceneInQueue();
271 }
272
273 private void OnClientSceneEvents(SceneEvent sceneEvent)
274 {
275 // if we have the scene loaded client-side it cannot be a network synched scene!
276 if (m_ClientLoader.IsSceneLoaded(sceneEvent.SceneName))
277 return;
278
279 // add or remove the scene from the synched scenes list
280 // we need these on client-side to unload them when client leaves
281 switch (sceneEvent.SceneEventType)
282 {
283 case SceneEventType.LoadComplete:
284 //Debug.Log($"Client loaded sync scene: {sceneEvent.SceneName}");
285 m_SynchedScenes.Add(new SceneReference(sceneEvent.Scene));
286 break;
287 case SceneEventType.UnloadComplete:
288 //Debug.Log($"Client unloaded sync scene: {sceneEvent.SceneName}");
289 m_SynchedScenes.Remove(new SceneReference(sceneEvent.Scene));
290 break;
291 }
