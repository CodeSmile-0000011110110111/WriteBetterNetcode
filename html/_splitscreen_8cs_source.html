59 internal void UpdateSplitscreen(CouchPlayers couchPlayers)
60 {
61 var firstTwoPlayers = AreOnlyFirstTwoPlayersPlaying(couchPlayers);
62 var playerCount = couchPlayers.PlayerCount;
63 if ((playerCount <= 1 || m_SplitscreenAxis == SplitscreenAxis.Disabled) &&
64 m_SplitscreenAxis != SplitscreenAxis.HorizontalAndVertical)
65 EnableSinglePlayerCamera();
66 else if (firstTwoPlayers && m_SplitscreenAxis == SplitscreenAxis.Horizontal)
67 EnableHorizontalSplitscreen();
68 else if (firstTwoPlayers && m_SplitscreenAxis == SplitscreenAxis.Vertical)
69 EnableVerticalSplitscreen();
70 else // 4-way split if set to both H/V, automatically for 3+ players, or if players 3 or 4 remain ingame
71 EnableFourWaySplitscreen();
72 }

80 private Boolean AreOnlyFirstTwoPlayersPlaying(CouchPlayers couchPlayers) =>

112 private void EnableFourWaySplitscreen()
113 {
114 SetAllPlayerCamerasActive();
115
116 // 0 = top left, 1 = top right, 2 = bottom left, 3 = bottom right
117 var cameras = m_Cameras.PlayerSplitCameras;
118 var viewportSize = new Vector2(0.5f, 0.5f);
119 cameras[0].rect = new Rect(new Vector2(0f, 0.5f), viewportSize);
120 cameras[1].rect = new Rect(new Vector2(0.5f, 0.5f), viewportSize);
121 cameras[2].rect = new Rect(new Vector2(0f, 0f), viewportSize);
122 cameras[3].rect = new Rect(new Vector2(0.5f, 0f), viewportSize);
123 }

125 private void SetSingleplayerCameraActive()
126 {
127 var playerCameras = m_Cameras.PlayerSplitCameras;
128 var notJoinedCameras = m_Cameras.PlayerNotJoinedCinecams;
129 for (var playerIndex = 0; playerIndex < Constants.MaxCouchPlayers; playerIndex++)
130 {
131 playerCameras[playerIndex].gameObject.SetActive(playerIndex == 0);
132 notJoinedCameras[playerIndex].gameObject.SetActive(false);
133 }
134 }
135
136 private void SetTwoPlayerCamerasActive()
137 {
138 var playerCameras = m_Cameras.PlayerSplitCameras;
139 var notJoinedCameras = m_Cameras.PlayerNotJoinedCinecams;
140 for (var playerIndex = 0; playerIndex < Constants.MaxCouchPlayers; playerIndex++)
141 {
142 playerCameras[playerIndex].gameObject.SetActive(playerIndex == 0 || playerIndex == 1);
143 notJoinedCameras[playerIndex].gameObject.SetActive(false);
144 }
145 }
146
147 private void SetAllPlayerCamerasActive()
148 {
149 var couchPlayers = ComponentsRegistry.Get<CouchPlayers>();
150 var playerCameras = m_Cameras.PlayerSplitCameras;
151 var notJoinedCameras = m_Cameras.PlayerNotJoinedCinecams;
152 for (var playerIndex = 0; playerIndex < Constants.MaxCouchPlayers; playerIndex++)
153 {
154 playerCameras[playerIndex].gameObject.SetActive(true);
155
156 var isPlaying = couchPlayers[playerIndex] != null;
157 notJoinedCameras[playerIndex].gameObject.SetActive(!isPlaying);
158 }
159 }
